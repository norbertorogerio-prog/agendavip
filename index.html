// ==================== VISÃO MOBILE (SEM REDUNDÂNCIA) ==================== function renderMobileView() { const container = document.getElementById("mobileView"); if (!container) return; const hoje = formatDateLocal(new Date()); // Eventos críticos: CHG (todos os futuros) const eventosCriticos = events.filter(e => e.tipo === "chg" && e.start >= hoje ).sort((a,b) => a.start.localeCompare(b.start)); // Próximos eventos (próximos 7 dias, INCLUINDO HOJE) const proximosEventos = []; for (let i = 0; i <= 7; i++) { const data = formatDateLocal(new Date(new Date().setDate(new Date().getDate() + i))); const eventosData = events.filter(e => occursOnDay(e, data)); proximosEventos.push(...eventosData); } // Remove duplicatas (eventos que duram mais de um dia) const proximosUnicos = [...new Map(proximosEventos.map(e => [e.id, e])).values()] .sort((a,b) => a.start.localeCompare(b.start)); let html = ''; // Seção: Eventos Críticos (CHG) if (eventosCriticos.length > 0) { html += `

⚠️ Eventos Críticos (CHG)

${eventosCriticos.map(ev => `
${formatDateTime(ev.start)}
${ev.title}
${ev.tipo} • ${ev.time || "Dia inteiro"}
`).join('')}
`; } // Seção: Próximos Dias (inclui hoje e próximos 7 dias) if (proximosUnicos.length > 0) { html += `

📅 Próximos 7 Dias

${proximosUnicos.slice(0, 10).map(ev => `
${formatDateTime(ev.start)}
${ev.title}
${ev.tipo} • ${ev.time || "Dia inteiro"}
`).join('')}
`; } if (eventosCriticos.length === 0 && proximosUnicos.length === 0) { html = `
✨ Nenhum compromisso agendado
Toque no botão + para criar
`; } container.innerHTML = html; }
